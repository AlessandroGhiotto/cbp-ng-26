\section{Introduction \& Objectives}
\begin{frame}{Introduction \& Objectives}
    \begin{block}{The Importance of Branch Prediction}
        \begin{itemize}
            \item Modern out-of-order superscalar processors execute instructions speculatively.
            \item Branch mispredictions flush the pipeline $\rightarrow$ severe \textbf{IPC penalty} and \textbf{energy waste}.
        \end{itemize}
    \end{block}
    \begin{block}{Project Objective}
        \begin{itemize}
            \item Evaluate predictor trade-offs under strict hardware memory budgets (\textbf{8 KB} and \textbf{16 KB}).
            \item Maximize the \textbf{Voltage-Frequency-Scaled Speedup (VFS)} metric:
            \begin{equation*}
                \text{VFS} = f(\text{Accuracy [MPKI]}, \text{Timing/Latency}, \text{Energy Per Instruction [EPI]})
            \end{equation*}
            \item Use the \textbf{HARCOM} physical model for cycle-accurate latency, area, and power estimation.
        \end{itemize}
    \end{block}
\end{frame}

\section{Block-Level Prediction}
\begin{frame}{Block-Level Prediction: Boosting IPC \& Energy Amortization}
    \begin{columns}[T]
        \column{0.55\textwidth}
        \begin{block}{Amortized SRAM Row Accesses}
            \begin{itemize}
                \item \textbf{Scalar Predictors:} Read/write SRAM tables for \emph{every single instruction} (fetch limit: 1 instr/cycle $\rightarrow$ IPC < 1.0).
                \item \textbf{Block Predictors (suffix L):} Predict an entire block of instructions (up to $LI=16$) in a single cycle by reading a wide SRAM row once.
                \item Subsequent branch predictions reuse active row bits via fast registers, skipping SRAM lookups.
            \end{itemize}
        \end{block}
        \column{0.41\textwidth}
        \begin{block}{Quantitative Benefits}
            \begin{itemize}
                \item \textbf{IPC Boost:} Blocks achieve an IPC range of 3.0--4.85 (vs. 0.49--0.98 for scalar).
                \item \textbf{Energy Reduction:} Amortization lowers EPI:
                \begin{itemize}
                    \item Scalar \texttt{gshare}: 263.2 fJ
                    \item Block \texttt{gshareL}: \textbf{145.5 fJ} (nearly half!)
                \end{itemize}
                \item Translates to a \textbf{3.0x to 5.64x VFS speedup} over scalar.
            \end{itemize}
        \end{block}
    \end{columns}
\end{frame}

\section{The TAGE Predictor \& Energy Challenges}
\begin{frame}{The TAGE Predictor \& Energy Challenges}
    \begin{columns}[T]
        \column{0.55\textwidth}
        \begin{block}{TAGE Architecture}
            \begin{itemize}
                \item State-of-the-art predictor using tagged tables indexed with geometrically increasing history lengths.
                \item Effectively captures long-term path correlations and short-term histories.
                \item Refined table update rules and usefulness ($u$) tracking.
                \item Achieves the lowest misprediction rate: \textbf{4.19 MPKI} for \texttt{tage\_uL}.
            \end{itemize}
        \end{block}
        \column{0.41\textwidth}
        \begin{block}{The Energy Bottleneck}
            \begin{itemize}
                \item Resolving predictions requires parallel read accesses to multiple tagged SRAM tables ($t_1, t_2, t_3$) in Stage 2.
                \item \textbf{High Dynamic Energy:} \texttt{tageL} EPI reaches \textbf{1145.5 fJ} (nearly 8x higher than \texttt{gshareL}).
            \end{itemize}
        \end{block}
    \end{columns}
\end{frame}

\section{Proposed Methodologies: Exploring the Design Space}
\begin{frame}{Proposed Methodologies: Exploring the Design Space}
    To mitigate TAGE's high energy overhead, we developed and evaluated three block-level TAGE variants:
    \vspace{0.2cm}
    \begin{enumerate}
        \item \textbf{TAGE bimode (\texttt{tage\_bimodeL})}
        \begin{itemize}
            \item Replaces the bimodal base with a choice-directed Bi-Mode predictor to reduce base-table aliasing.
            \item \emph{Limitation:} Wide base tables consume 64\% of storage. Squeezed tagged tables must use short 8-bit tags, causing severe tag aliasing (\textbf{9.74 MPKI}) and high EPI (\textbf{1587.4 fJ}).
        \end{itemize}
        \vspace{0.1cm}
        \item \textbf{TAGE bias (\texttt{tage\_biasL})}
        \begin{itemize}
            \item Integrates a small dedicated bias table (from Bi-Mode++) to predict strongly biased branches early.
            \item \emph{Limitation:} Achieves great accuracy (\textbf{5.25 MPKI}) and lower energy (\textbf{932.8 fJ}), but sequential dependency in Stage 1 stretches the critical path past 300 ps, doubling latency $\rightarrow$ IPC drops to \textbf{2.83}.
        \end{itemize}
    \end{enumerate}
\end{frame}

\section{Our Final Proposal: TAGE sat}
\begin{frame}{Our Final Proposal: TAGE sat (Saturation Bypass)}
    \begin{block}{Concept: Saturation-Based Bypassing}
        \begin{itemize}
            \item Reuses bimodal base counters: saturated counter (0 or max) $\rightarrow$ strongly biased.
            \item Dynamically bypasses TAGE tagged lookups ($t_1, t_2, t_3$) using the HARCOM \texttt{execute\_if} primitive.
        \end{itemize}
    \end{block}
    \begin{columns}[T]
        \column{0.48\textwidth}
        \textbf{Throughput Gains}
        \begin{itemize}
            \item Bypassing avoids Stage 2 RAM lookup latency.
            \item Predictions resolve quickly from registers.
            \item Achieves the highest IPC: \textbf{4.85} (16 KB).
        \end{itemize}
        \column{0.48\textwidth}
        \textbf{Energy Reductions}
        \begin{itemize}
            \item Bypassing prevents power-hungry RAM reads.
            \item Dynamic energy (EPI) drops by \textbf{14.6\%} (16 KB) and \textbf{21.3\%} (8 KB) vs. \texttt{tageL}.
        \end{itemize}
    \end{columns}
\end{frame}

\section{Conclusions \& Future Work}
\begin{frame}{Conclusions \& Future Work}
    \begin{block}{Conclusions}
        \begin{itemize}
            \item Block-level (line-wide) prediction is crucial: boosts IPC by 7x and reduces EPI by half via SRAM read amortization.
            \item Saturation-based bypass (\texttt{tage\_satL}) is highly effective: achieves the highest IPC (4.85) and reduces TAGE energy consumption by 14.6\%--21.3\% with competitive VFS efficiency.
            \item High predictor complexity (e.g., in \texttt{tage\_biasL} and \texttt{tage\_bimodeL}) degrades performance due to timing path violations or storage budget conflicts.
        \end{itemize}
    \end{block}
    \begin{block}{Future Work}
        \begin{itemize}
            \item Implement a collection of smaller, specialized predictors targeted at specific branch behaviors.
            \item Dynamically steer branches to the simplest predictor that can handle them, avoiding the energy overhead of monolithic TAGE tables.
            \end{itemize}
    \end{block}
\end{frame}
